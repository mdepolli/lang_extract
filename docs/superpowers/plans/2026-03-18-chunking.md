# Chunking Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add document chunking with sentence-aware splitting and integrate it into the orchestrator for processing long documents via `Task.async_stream`.

**Architecture:** `Chunker` tokenizes the source, detects sentence boundaries, then packs sentences into chunks within a character budget. Three tiers: sentence packing → newline splitting → token fallback. The orchestrator processes chunks in parallel via `Task.async_stream`, adjusting byte offsets on the resulting spans.

**Tech Stack:** Elixir, ExUnit, HTTPower.Test (for orchestrator integration tests)

**Spec:** `docs/superpowers/specs/2026-03-18-chunking-design.md`

```elixir
# lib/lang_extract/chunker.ex
defmodule LangExtract.Chunker do
  @moduledoc """
  Splits text into chunks respecting sentence boundaries.

  Uses the existing tokenizer for sentence detection and provides
  three-tier chunking: sentence packing, newline splitting, token fallback.
  """

  alias LangExtract.Alignment.Tokenizer

  @known_abbreviations MapSet.new(["Mr.", "Mrs.", "Ms.", "Dr.", "Prof.", "St."])
  @sentence_enders MapSet.new([".", "!", "?"])
  @closing_punctuation MapSet.new(["\"", "'", "\u201D", "\u2019", "\u00BB", ")", "]", "}"])

  defmodule Chunk do
    @moduledoc """
    A text chunk with its byte position in the source.
    """

    @type t :: %__MODULE__{
            text: String.t(),
            byte_start: non_neg_integer()
          }

    @enforce_keys [:text, :byte_start]
    defstruct [:text, :byte_start]
  end

  @doc """
  Splits text into sentence strings.

  Sentences are detected by walking tokens and checking for sentence-ending
  punctuation (`.`, `!`, `?`), with abbreviation awareness and newline heuristics.
  """
  @spec find_sentences(String.t()) :: [String.t()]
  def find_sentences(""), do: []

  def find_sentences(text) when is_binary(text) do
    tokens = Tokenizer.tokenize(text)

    if tokens == [] do
      []
    else
      boundaries = find_sentence_boundaries(text, tokens)
      extract_sentence_texts(text, tokens, boundaries)
    end
  end

  defp find_sentence_boundaries(text, tokens) do
    token_count = length(tokens)

    tokens
    |> Enum.with_index()
    |> Enum.reduce([], fn {token, idx}, boundaries ->
      cond do
        # Rule 1: Sentence-ending punctuation (not abbreviation)
        token.type == :punctuation and MapSet.member?(@sentence_enders, token.text) and
            not abbreviation?(tokens, idx) ->
          # Consume trailing closing punctuation
          end_idx = consume_closing(tokens, idx + 1, token_count)
          [end_idx | boundaries]

        # Rule 3: Newline + uppercase
        token.type == :whitespace and String.contains?(token.text, "\n") and
            next_starts_uppercase?(text, tokens, idx, token_count) ->
          [idx | boundaries]

        true ->
          boundaries
      end
    end)
    |> Enum.reverse()
    |> Enum.uniq()
  end

  defp abbreviation?(tokens, idx) do
    if idx > 0 do
      prev = Enum.at(tokens, idx - 1)
      prev.type == :word and MapSet.member?(@known_abbreviations, "#{prev.text}.")
    else
      false
    end
  end

  defp consume_closing(tokens, idx, token_count) do
    if idx < token_count do
      token = Enum.at(tokens, idx)

      if token.type == :punctuation and MapSet.member?(@closing_punctuation, token.text) do
        consume_closing(tokens, idx + 1, token_count)
      else
        idx
      end
    else
      idx
    end
  end

  defp next_starts_uppercase?(text, tokens, idx, token_count) do
    if idx + 1 < token_count do
      next = Enum.at(tokens, idx + 1)

      if next.type in [:word, :number] do
        first_char = binary_part(text, next.byte_start, 1)
        first_char == String.upcase(first_char) and first_char != String.downcase(first_char)
      else
        false
      end
    else
      false
    end
  end

  defp extract_sentence_texts(text, tokens, boundaries) do
    token_count = length(tokens)
    # Add boundaries at start and end
    starts = [0 | boundaries]
    ends = boundaries ++ [token_count]

    Enum.zip(starts, ends)
    |> Enum.reject(fn {s, e} -> s == e end)
    |> Enum.map(fn {start_idx, end_idx} ->
      first = Enum.at(tokens, start_idx)
      last = Enum.at(tokens, end_idx - 1)
      binary_part(text, first.byte_start, last.byte_end - first.byte_start)
    end)
  end
end
```

```elixir
  @doc """
  Splits text into chunks respecting sentence boundaries.

  ## Options

    * `:max_chunk_size` — maximum characters per chunk (required)

  """
  @spec chunk(String.t(), keyword()) :: [Chunk.t()]
  def chunk("", _opts), do: []

  def chunk(text, opts) when is_binary(text) do
    max_size = Keyword.fetch!(opts, :max_chunk_size)
    tokens = Tokenizer.tokenize(text)
    sentences = find_sentences_from_tokens(text, tokens)

    pack_sentences(text, sentences, max_size)
  end

  # Returns list of {byte_start, byte_end} for each sentence
  defp find_sentences_from_tokens(_text, []), do: []

  defp find_sentences_from_tokens(text, tokens) do
    boundaries = find_sentence_boundaries(text, tokens)
    token_count = length(tokens)
    starts = [0 | boundaries]
    ends = boundaries ++ [token_count]

    Enum.zip(starts, ends)
    |> Enum.reject(fn {s, e} -> s == e end)
    |> Enum.map(fn {start_idx, end_idx} ->
      first = Enum.at(tokens, start_idx)
      last = Enum.at(tokens, end_idx - 1)
      {first.byte_start, last.byte_end}
    end)
  end

  defp pack_sentences(text, sentences, max_size) do
    sentences
    |> Enum.reduce({[], nil}, fn {sent_start, sent_end}, {chunks, current} ->
      sent_size = sent_end - sent_start

      case current do
        nil ->
          {chunks, {sent_start, sent_end}}

        {chunk_start, chunk_end} ->
          combined_size = sent_end - chunk_start

          if combined_size <= max_size do
            {chunks, {chunk_start, sent_end}}
          else
            chunk = make_chunk(text, chunk_start, chunk_end)
            {[chunk | chunks], {sent_start, sent_end}}
          end
      end
    end)
    |> then(fn
      {chunks, nil} -> Enum.reverse(chunks)
      {chunks, {start, end_}} -> Enum.reverse([make_chunk(text, start, end_) | chunks])
    end)
  end

  defp make_chunk(text, byte_start, byte_end) do
    %Chunk{
      text: binary_part(text, byte_start, byte_end - byte_start),
      byte_start: byte_start
    }
  end
```

```elixir
defmodule LangExtract.Orchestrator do
  @moduledoc """
  Wires the full extraction pipeline.

  Builds a prompt, calls the LLM provider, normalizes and parses the response,
  aligns extractions to source text, and returns enriched spans.

  When `:max_chunk_size` is set, splits the source into chunks and processes
  them in parallel via `Task.async_stream`.
  """

  alias LangExtract.{
    Alignment.Aligner,
    Alignment.Span,
    Chunker,
    Client,
    Extraction,
    FormatHandler,
    Parser,
    Prompt
  }

  @spec run(Client.t(), String.t(), Prompt.Template.t(), keyword()) ::
          {:ok, [Span.t()]} | {:error, term()}
  def run(%Client{} = client, source, %Prompt.Template{} = template, opts \\ []) do
    case Keyword.get(opts, :max_chunk_size) do
      nil -> run_single(client, source, template, opts)
      max_size -> run_chunked(client, source, template, max_size, opts)
    end
  end

  defp run_single(client, source, template, opts) do
    prompt = Prompt.Builder.build(template, source)

    with {:ok, raw_output} <- client.provider.infer(prompt, client.options),
         {:ok, normalized} <- FormatHandler.normalize(raw_output),
         {:ok, extractions} <- Parser.parse(normalized) do
      {:ok, enrich(extractions, source, opts)}
    end
  end

  defp run_chunked(client, source, template, max_size, opts) do
    chunks = Chunker.chunk(source, max_chunk_size: max_size)

    if chunks == [] do
      {:ok, []}
    else
      max_concurrency = Keyword.get(opts, :max_concurrency, 3)

      chunks
      |> Enum.with_index()
      |> Task.async_stream(
        fn {chunk, idx} ->
          prev_text = if idx > 0, do: Enum.at(chunks, idx - 1).text
          process_chunk(client, chunk, template, prev_text, opts)
        end,
        ordered: true,
        max_concurrency: max_concurrency
      )
      |> Enum.reduce_while({:ok, []}, fn
        {:ok, {:ok, spans}}, {:ok, acc} ->
          {:cont, {:ok, acc ++ spans}}

        {:ok, {:error, _} = error}, _acc ->
          {:halt, error}

        {:exit, reason}, _acc ->
          {:halt, {:error, {:task_error, reason}}}
      end)
    end
  end

  defp process_chunk(client, chunk, template, prev_text, opts) do
    builder_opts = if prev_text, do: [previous_chunk: prev_text], else: []
    prompt = Prompt.Builder.build(template, chunk.text, builder_opts)

    with {:ok, raw_output} <- client.provider.infer(prompt, client.options),
         {:ok, normalized} <- FormatHandler.normalize(raw_output),
         {:ok, extractions} <- Parser.parse(normalized) do
      spans = enrich(extractions, chunk.text, opts)
      adjusted = adjust_offsets(spans, chunk.byte_start)
      {:ok, adjusted}
    end
  end

  defp enrich(extractions, source, opts) do
    texts = Enum.map(extractions, & &1.text)
    spans = Aligner.align(source, texts, opts)

    Enum.zip(extractions, spans)
    |> Enum.map(fn {%Extraction{} = extraction, %Span{} = span} ->
      %Span{span | class: extraction.class, attributes: extraction.attributes}
    end)
  end

  defp adjust_offsets(spans, byte_offset) do
    Enum.map(spans, fn
      %Span{byte_start: nil} = span -> span
      %Span{byte_start: bs, byte_end: be} = span ->
        %Span{span | byte_start: bs + byte_offset, byte_end: be + byte_offset}
    end)
  end
end
```
